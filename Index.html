<script>
pdfjsLib.GlobalWorkerOptions.workerSrc = 'https://cdnjs.cloudflare.com/ajax/libs/pdf.js/3.11.174/pdf.worker.min.js';

const COLORS = [
  { bg:'#FBBC04', fg:'#000', fill:'rgba(251,188,4,0.4)',  stroke:'#f09d00' },
  { bg:'#EA4335', fg:'#fff', fill:'rgba(234,67,53,0.35)', stroke:'#c5221f' },
  { bg:'#34A853', fg:'#fff', fill:'rgba(52,168,83,0.35)', stroke:'#1e8e3e' },
  { bg:'#4285F4', fg:'#fff', fill:'rgba(66,133,244,0.35)',stroke:'#1967d2' },
  { bg:'#FF6D00', fg:'#fff', fill:'rgba(255,109,0,0.35)', stroke:'#e65100' },
  { bg:'#9C27B0', fg:'#fff', fill:'rgba(156,39,176,0.35)',stroke:'#6a0080' },
];

const $ = id => document.getElementById(id);
let pdfDoc = null, keywords = [];
let pages = []; // {container, canvas, overlay, vp, renderScale}
let textCache = {};
let allMatches = []; // {page(0-indexed), ki, boxes:[{x,y,w,h}]}
let kwMatchIdx = []; // kwMatchIdx[ki] = [globalIdx, ...]
let kwCur = [];
let globalCur = -1;
let panelCollapsed = false;

// ─── File load
$('file-input').addEventListener('change', e => { if (e.target.files[0]) loadFile(e.target.files[0]); });

async function loadFile(file) {
  showStatus('Loading…');
  const buf = await file.arrayBuffer();
  pdfDoc = await pdfjsLib.getDocument({ data: buf }).promise;
  textCache = {}; allMatches = []; kwMatchIdx = []; kwCur = [];
  keywords = []; renderTags();
  $('upload-screen').style.display = 'none';
  $('viewer').style.display = 'flex';
  $('panel').style.display = 'flex';
  $('kw-results').innerHTML = ''; $('kw-results').style.display = 'none';
  $('global-row').style.display = 'none';
  updateSearchBtn();
  await renderAllPages();
  hideStatus();
}

async function renderAllPages() {
  const scroll = $('pdf-scroll');
  scroll.innerHTML = '';
  pages = [];
  const maxW = Math.min(window.innerWidth - 4, 1200);

  for (let p = 1; p <= pdfDoc.numPages; p++) {
    const page = await pdfDoc.getPage(p);
    const baseVP = page.getViewport({ scale: 1 });
    const scale  = (maxW) / baseVP.width;
    const vp     = page.getViewport({ scale });

    const container = document.createElement('div');
    container.className = 'page-container';
    container.style.width  = vp.width  + 'px';
    container.style.height = vp.height + 'px';

    const canvas = document.createElement('canvas');
    canvas.width  = vp.width;
    canvas.height = vp.height;

    const overlay = document.createElement('div');
    overlay.className = 'hl-overlay';
    overlay.style.width  = vp.width  + 'px';
    overlay.style.height = vp.height + 'px';

    container.appendChild(canvas);
    container.appendChild(overlay);
    scroll.appendChild(container);

    await page.render({ canvasContext: canvas.getContext('2d'), viewport: vp }).promise;
    pages.push({ container, canvas, overlay, vp, scale, pageNum: p });

    $('prog').style.width = `${Math.round(p / pdfDoc.numPages * 100)}%`;
  }
  $('prog').style.width = '0';
}

// ─── Text extraction with CORRECT coords
async function getItems(pageNum) {
  if (textCache[pageNum]) return textCache[pageNum];
  const page = await pdfDoc.getPage(pageNum);
  const tc = await page.getTextContent({ includeMarkedContent: false });

  // Use the SAME viewport as rendering so coords match
  const pg = pages[pageNum - 1];
  const vp = pg.vp;

  const items = [];
  for (const item of tc.items) {
    if (!item.str) continue;
    // PDF.js provides transform in PDF space; apply viewport transform to get canvas coords
    const tx = pdfjsLib.Util.transform(vp.transform, item.transform);
    // tx[4] = x, tx[5] = y (these are already in canvas pixel coords for this viewport)
    const x = tx[4];
    const y = tx[5];
    // item.width and item.height are in PDF user space units — scale them
    const w = item.width  * pg.scale;
    const h = item.height * pg.scale;
    // y in canvas coords is baseline; go up by height to get top
    items.push({ str: item.str, x, y: y - h, w, h });
  }
  textCache[pageNum] = items;
  return items;
}

// ─── Keywords
$('kw-add-btn').addEventListener('click', addKW);
$('kw-input').addEventListener('keydown', e => { if (e.key === 'Enter') addKW(); });

function addKW() {
  const v = $('kw-input').value.trim();
  if (!v || keywords.find(k => k.text.toLowerCase() === v.toLowerCase())) return;
  keywords.push({ text: v, color: COLORS[keywords.length % COLORS.length] });
  $('kw-input').value = '';
  renderTags(); updateSearchBtn();
}

function renderTags() {
  $('tags').innerHTML = '';
  keywords.forEach((kw, i) => {
    const t = document.createElement('span');
    t.className = 'tag';
    t.style.cssText = `background:${kw.color.bg};color:${kw.color.fg}`;
    t.innerHTML = `${esc(kw.text)}<span class="x" data-i="${i}"> ×</span>`;
    $('tags').appendChild(t);
  });
  $('tags').querySelectorAll('.x').forEach(x => x.addEventListener('click', () => {
    keywords.splice(+x.dataset.i, 1); renderTags(); updateSearchBtn();
  }));
}

function updateSearchBtn() { $('btn-search').disabled = !pdfDoc || !keywords.length; }

// ─── Search
$('btn-search').addEventListener('click', runSearch);

async function runSearch() {
  allMatches = [];
  kwMatchIdx = keywords.map(() => []);
  kwCur = keywords.map(() => -1);
  clearAllHL();
  showStatus('Searching…');

  const cs = $('opt-case').checked;
  const ww = $('opt-whole').checked;

  for (let p = 1; p <= pdfDoc.numPages; p++) {
    const items = await getItems(p);

    keywords.forEach((kw, ki) => {
      const re = mkRe(kw.text, cs, ww);
      // Search item by item for accurate per-item bounding boxes
      items.forEach(item => {
        let m;
        const itemRe = mkRe(kw.text, cs, ww);
        while ((m = itemRe.exec(item.str)) !== null) {
          // Fraction of item width for this match
          const charStart = m.index / item.str.length;
          const charEnd   = (m.index + m[0].length) / item.str.length;
          const x = item.x + charStart * item.w;
          const w = (charEnd - charStart) * item.w;
          const box = { x, y: item.y, w: Math.max(w, 6), h: item.h };
          const idx = allMatches.length;
          allMatches.push({ pageIdx: p - 1, ki, box });
          kwMatchIdx[ki].push(idx);
        }
      });
    });

    $('prog').style.width = `${Math.round(p / pdfDoc.numPages * 100)}%`;
  }
  $('prog').style.width = '0';

  drawAllHL(-1);
  buildResultRows();
  hideStatus();

  if (allMatches.length) {
    globalCur = 0;
    const m = allMatches[0];
    kwCur[m.ki] = kwMatchIdx[m.ki].indexOf(0);
    drawAllHL(globalCur);
    scrollToMatch(globalCur);
    $('global-row').style.display = 'flex';
    updateCounters();
  }

  collapsePanel();
}

function mkRe(kw, cs, ww) {
  let p = kw.replace(/[.*+?^${}()|[\]\\]/g, '\\$&');
  if (ww) p = `\\b${p}\\b`;
  return new RegExp(p, cs ? 'g' : 'gi');
}

// ─── Highlight rendering (DOM divs — pixel-perfect)
function clearAllHL() {
  pages.forEach(pg => { pg.overlay.innerHTML = ''; });
}

function drawAllHL(activeidx) {
  clearAllHL();
  allMatches.forEach((m, idx) => {
    const kw = keywords[m.ki];
    const pg = pages[m.pageIdx];
    const b  = m.box;
    const div = document.createElement('div');
    div.className = 'hl-rect';
    const isActive = idx === activeidx;
    div.style.cssText = `
      left:${b.x}px; top:${b.y}px;
      width:${b.w}px; height:${b.h}px;
      background:${isActive ? kw.color.fill.replace(/[\d.]+\)/, '0.7)') : kw.color.fill};
      outline:${isActive ? `2.5px solid ${kw.color.stroke}` : 'none'};
    `;
    pg.overlay.appendChild(div);
  });
}

function scrollToMatch(idx) {
  if (idx < 0 || idx >= allMatches.length) return;
  const m  = allMatches[idx];
  const pg = pages[m.pageIdx];
  const scroll = $('pdf-scroll');
  const top = pg.container.offsetTop + m.box.y - window.innerHeight * 0.38;
  scroll.scrollTo({ top: Math.max(0, top), behavior: 'smooth' });
}

// ─── Result rows
function buildResultRows() {
  const el = $('kw-results');
  el.innerHTML = '';
  if (!allMatches.length) { el.style.display = 'none'; return; }
  el.style.display = 'block';

  keywords.forEach((kw, ki) => {
    const cnt = kwMatchIdx[ki].length;
    const row = document.createElement('div');
    row.className = 'kw-result-row';
    row.innerHTML = `
      <span class="kw-chip" style="background:${kw.color.bg};color:${kw.color.fg}" title="${esc(kw.text)}">${esc(kw.text)}</span>
      <span class="kw-found">${cnt} found</span>
      <div class="kw-nav">
        <button onclick="kwNav(${ki},-1)">↑</button>
        <span class="kw-pos" id="kwpos-${ki}">${cnt ? `0/${cnt}` : '—'}</span>
        <button onclick="kwNav(${ki},1)">↓</button>
      </div>`;
    if (!cnt) { row.querySelector('.kw-nav button').disabled = true; row.querySelectorAll('.kw-nav button')[1].disabled = true; }
    el.appendChild(row);
  });
}

function kwNav(ki, dir) {
  const list = kwMatchIdx[ki];
  if (!list.length) return;
  kwCur[ki] = kwCur[ki] < 0 ? 0 : (kwCur[ki] + dir + list.length) % list.length;
  globalCur = list[kwCur[ki]];
  drawAllHL(globalCur);
  scrollToMatch(globalCur);
  updateCounters();
}

function globalNav(dir) {
  if (!allMatches.length) return;
  globalCur = (globalCur + dir + allMatches.length) % allMatches.length;
  const m = allMatches[globalCur];
  kwCur[m.ki] = kwMatchIdx[m.ki].indexOf(globalCur);
  drawAllHL(globalCur);
  scrollToMatch(globalCur);
  updateCounters();
}

function updateCounters() {
  keywords.forEach((kw, ki) => {
    const el = $(`kwpos-${ki}`);
    if (!el) return;
    const cnt = kwMatchIdx[ki].length;
    if (!cnt) { el.textContent = '—'; return; }
    const pos = kwCur[ki];
    el.textContent = pos >= 0 ? `${pos + 1}/${cnt}` : `1/${cnt}`;
  });
  if (allMatches.length) {
    $('global-counter').textContent = `${globalCur + 1}/${allMatches.length}`;
    const kw = keywords[allMatches[globalCur].ki];
    $('global-label').innerHTML = `<span style="color:${kw.color.stroke};font-weight:700">${esc(kw.text)}</span>`;
  }
}

// ─── Panel
function expandPanel() {
  panelCollapsed = false;
  $('panel').classList.remove('collapsed');
  $('btn-coll').textContent = '▲';
}
function collapsePanel() {
  panelCollapsed = true;
  $('panel').classList.add('collapsed');
  $('btn-coll').textContent = '▼';
}
function togglePanel() {
  panelCollapsed ? expandPanel() : collapsePanel();
}

function newPDF() {
  $('upload-screen').style.display = 'flex';
  $('viewer').style.display = 'none';
  $('panel').style.display = 'none';
  $('pdf-scroll').innerHTML = '';
  pdfDoc = null; keywords = []; textCache = {};
  allMatches = []; kwMatchIdx = []; kwCur = [];
  $('kw-results').innerHTML = ''; $('kw-results').style.display = 'none';
  $('global-row').style.display = 'none';
  renderTags(); updateSearchBtn();
  panelCollapsed = false;
}

function showStatus(msg) { const s = $('status'); s.textContent = msg; s.style.display = 'block'; }
function hideStatus() { $('status').style.display = 'none'; }
function esc(s) { return String(s).replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;'); }
</script>
